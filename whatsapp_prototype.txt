import pandas as pd
import pywhatkit
import datetime as dt
import time
import schedule

CASES_FILE = "cases.csv"
INDIAN_TZ_OFFSET = 5.5  # if needed later

def load_cases():
    df = pd.read_csv(CASES_FILE)
    # ensure date/time columns are strings
    df["hearing_date"] = df["hearing_date"].astype(str)
    df["hearing_time"] = df["hearing_time"].astype(str)
    return df

def send_whatsapp_message(phone: str, message: str):
    print(f"Sending to {phone}: {message}")
    # instantly sends via web.whatsapp.com (must be logged in)
    pywhatkit.sendwhatmsg_instantly(
        phone_no=phone,
        message=message,
        wait_time=20,   # seconds to wait while opening WhatsApp Web
        tab_close=True,
        close_time=3
    )
    # small sleep to avoid spamming too fast
    time.sleep(5)

def send_tomorrow_reminders():
    today = dt.date.today()
    tomorrow = today + dt.timedelta(days=1)
    tomorrow_str = tomorrow.strftime("%Y-%m-%d")

    df = load_cases()
    # filter cases with hearing_date == tomorrow
    upcoming = df[df["hearing_date"] == tomorrow_str]

    if upcoming.empty:
        print("No hearings tomorrow.")
        return

    for _, row in upcoming.iterrows():
        client = row["client_name"]
        phone = row["phone"]
        case_id = row["case_id"]
        hearing_date = row["hearing_date"]
        hearing_time = row["hearing_time"]

        msg = (
            f"Dear {client},\n"
            f"Reminder: Your next hearing for Case {case_id} is scheduled on "
            f"{hearing_date} at {hearing_time}.\n\n"
            f"- Advocate Office"
        )

        send_whatsapp_message(phone, msg)

def main():
    # schedule the job every day at 09:00
    schedule.every().day.at("09:00").do(send_tomorrow_reminders)

    print("Scheduler started. Waiting for next run...")
    while True:
        schedule.run_pending()
        time.sleep(1)

if __name__ == "__main__":
    main()

import sqlite3
import pandas as pd

CSV_FILE = "cases.csv"
DB_FILE = "cases.db"

def import_data():
    df = pd.read_csv(CSV_FILE)

    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()

    for _, row in df.iterrows():
        client = row["client_name"]
        phone = str(row["phone"])
        if not phone.startswith("+"):
            phone = "+" + phone

        case_id = row["case_id"]
        hearing_date = str(row["hearing_date"])
        hearing_time = str(row["hearing_time"])

        cur.execute("""
        INSERT INTO cases (client_name, phone, case_id, hearing_date, hearing_time)
        VALUES (?, ?, ?, ?, ?)
        """, (client, phone, case_id, hearing_date, hearing_time))

    conn.commit()
    conn.close()
    print("CSV data imported into SQLite successfully.")

if __name__ == "__main__":
    import_data()

def send_all_reminders():
    today = dt.date.today()

    conn = sqlite3.connect("cases.db")
    cur = conn.cursor()

    # fetch all cases
    cur.execute("SELECT client_name, phone, case_id, hearing_date, hearing_time FROM cases")
    rows = cur.fetchall()

    for client, phone, case_id, hearing_date, hearing_time in rows:

        hearing_dt = dt.datetime.strptime(hearing_date, "%Y-%m-%d").date()

        delta = (hearing_dt - today).days

        # 2-days-before
        if delta == 2:
            msg = f"Reminder: Your hearing for Case {case_id} is in 2 days."
            send_whatsapp_message(phone, msg)

        # 1-day-before
        elif delta == 1:
            msg = f"Reminder: Your hearing for Case {case_id} is tomorrow at {hearing_time}."
            send_whatsapp_message(phone, msg)

        # same-day
        elif delta == 0:
            msg = f"Today is your hearing for Case {case_id} at {hearing_time}."
            send_whatsapp_message(phone, msg)

    conn.close()

import time
import sqlite3
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager

DB_FILE = "cases.db"

# --- DB Function ---
def search_case(query_text):
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()

    # if user types: case 12345
    if "case" in query_text:
        parts = query_text.split()
        case_id = None
        for p in parts:
            if p.isdigit():
                case_id = p
                break

        if case_id:
            cur.execute("SELECT client_name, hearing_date, hearing_time FROM cases WHERE case_id=?", (case_id,))
            row = cur.fetchone()
            conn.close()

            if row:
                name, date, time_ = row
                return f"Case {case_id}:\nClient: {name}\nNext hearing: {date} at {time_}"
            else:
                return "Case not found."

    # if user types: next hearing
    if "next hearing" in query_text:
        cur.execute("SELECT client_name, case_id, hearing_date, hearing_time FROM cases ORDER BY hearing_date LIMIT 1")
        row = cur.fetchone()
        conn.close()

        if row:
            name, case_id, date, time_ = row
            return f"Next hearing is:\nCase {case_id}\nClient: {name}\nDate: {date} at {time_}"
        else:
            return "No upcoming hearings."

    conn.close()
    return "I didn't understand. Try: 'case 12345' or 'next hearing'."


# --- WhatsApp Bot Function ---
def start_whatsapp_bot():

    print("Starting WhatsApp bot...")

    # Launch Chrome with driver manager
    driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))

    # Open WhatsApp Web
    driver.get("https://web.whatsapp.com")
    print("Please scan the QR code on WhatsApp Web.")
    time.sleep(15)

    print("Bot is now listening for messages...")

    last_text = ""

    while True:
        time.sleep(2)

        try:
            # Find all received messages (in the last chat)
            messages = driver.find_elements(By.XPATH, "//div[contains(@class,'message-in')]//div[contains(@class,'copyable-text')]")

            if not messages:
                continue

            latest_msg = messages[-1]
            msg_text = latest_msg.text.strip()

            if msg_text == last_text:
                continue  # ignore duplicates

            print("New message received:", msg_text)
            last_text = msg_text

            # Process message
            reply = search_case(msg_text.lower())

            # Message box
            input_box = driver.find_element(By.XPATH, "//div[@title='Type a message']")
            input_box.click()
            time.sleep(1)
            input_box.send_keys(reply)
            time.sleep(1)

            # Send button
            send_button = driver.find_element(By.XPATH, "//span[@data-icon='send']")
            send_button.click()

            print("Replied:", reply)

        except Exception as e:
            print("Error:", e)
            continue


if __name__ == "__main__":
    start_whatsapp_bot()
